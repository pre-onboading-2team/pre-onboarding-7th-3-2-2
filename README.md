### json-server 을 활용 전체 데이터 조회 기능 구현

```js

  useEffect(() => {
    fetch('http://localhost:4000/accounts?_expand=user', {
      method: 'GET',
      headers: {
        Authorization: 'Bearer ' + token,
      },
    })
      .then(response => response.json())
      .then(result => {
        setData(result);
      });
    console.log(token);
  }, []);
  
  ```
